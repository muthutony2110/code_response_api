from flask import Flask, request, jsonify
from filters.filters import is_greeting
from message import get_greeting_response, get_decline_message
from querychecker import classify_text_with_deepseek
from ollama_client import query_ollama, split_message_and_code, build_prompt_with_context
from mssql_client import (
    save_history_with_task,
    get_history_by_task,
    get_all_user_history,
    delete_user_history,
    delete_task_history
)

app = Flask(__name__)

@app.route("/generate", methods=["POST"])
def generate():
    try:
        data = request.get_json()
        prompt = data.get("prompt", "").strip()
        user_id = data.get("user_id", "guest").strip()
        task_name = data.get("task_name", "").strip()

        if not prompt or not task_name:
            return jsonify({
                "response": {
                    "message": "‚ö†Ô∏è Please provide both 'prompt' and 'task_name'.",
                    "code": ""
                }
            })

        # Greeting check
        if is_greeting(prompt):
            return jsonify({
                "response": {
                    "message": get_greeting_response(),
                    "code": ""
                }
            })

        # Classification
        classification = classify_text_with_deepseek(prompt)
        if classification == "Not coding-related":
            # Don't save non-coding prompts
            return jsonify({
                "response": {
                    "message": get_decline_message(),
                    "code": ""
                }
            })
        elif classification == "Unclear":
            # Don't save unclear prompts
            return jsonify({
                "response": {
                    "message": "ü§î I'm not sure if it's about programming. Please clarify.",
                    "code": ""
                }
            })

        # Fetch context for this user + task
        history = get_history_by_task(user_id, task_name, limit=3)
        full_prompt = build_prompt_with_context(history, prompt)

        # Query model
        raw_response = query_ollama(full_prompt)
        structured = split_message_and_code(raw_response)

        # Save only coding-related prompts
        save_history_with_task(user_id, task_name, prompt, structured['message'], structured['code'])

        return jsonify({"response": structured, "task_name": task_name})

    except Exception as e:
        import traceback
        traceback.print_exc()
        return jsonify({
            "response": {
                "message": f"‚ùå Internal Server Error: {str(e)}",
                "code": ""
            }
        }), 500

import subprocess
import re

def query_ollama(full_prompt):
    """
    Send a prompt to DeepSeek LLM via Ollama CLI and return its output.
    full_prompt: str - This should include the full context + latest prompt.
    """
    try:
        result = subprocess.run(
            ["ollama", "run", "deepseek-coder-v2:16b"],
            input=full_prompt.encode("utf-8"),
            capture_output=True,
            timeout=240
        )
        return result.stdout.decode("utf-8").strip()
    except subprocess.TimeoutExpired:
        return "‚è∞ DeepSeek took too long to respond."
    except Exception as e:
        return f"‚ùå Error calling DeepSeek: {str(e)}"


def build_prompt_with_context(history, current_prompt):
    """
    Combines past history and current prompt into one structured prompt for better LLM understanding.
    """
    base_instruction = (
        "You are a professional coding assistant.\n\n"
        "TASK:\n"
        "- If the user asks for code, provide the full code clearly.\n"
        "- If the user asks a conceptual or coding-related question, give a short and clear explanation.\n\n"
        "Respond ONLY if the topic is about programming or development.\n\n"
    )

    context = ""
    for item in history:
        context += f"User: {item['prompt']}\nAssistant: {item['message']}\n"

    full_prompt = f"{base_instruction}{context}User: {current_prompt}\nAssistant:"
    return full_prompt


def split_message_and_code(response_text):
    if not response_text:
        return {"message": "‚ö†Ô∏è No response from model.", "code": ""}

    code_match = re.search(r"```(?:[a-zA-Z]*\n)?(.*?)```", response_text, re.DOTALL)

    if code_match:
        return {
            "message": response_text.split("```")[0].strip(),
            "code": code_match.group(1).strip()
        }

    lines = response_text.splitlines()
    return {
        "message": "\n".join(lines[:3]).strip(),
        "code": "\n".join(lines[3:]).strip()
    }
